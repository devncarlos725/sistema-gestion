-- =============================================

-- 1. TABLA DE CLIENTES
-- Acá guardamos a los que nos compran. Metimos CUIT para ARCA 
-- y saldos en pesos/dolares para manejar la cuenta corriente (fiado).
CREATE TABLE clientes (
  id           UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  nombre       TEXT NOT NULL,
  cuit_cuil    TEXT, -- Importante para la factura A
  telefono     TEXT,
  email        TEXT,
  descuento_pct DECIMAL(5,2) DEFAULT 0, -- Si el cuñado le quiere cobrar menos a un amigo
  saldo_pesos  DECIMAL(15,2) DEFAULT 0, -- Lo que debe en pesos
  saldo_usd    DECIMAL(15,2) DEFAULT 0, -- Lo que debe en verdes (para no perder con la infla)
  activo       BOOLEAN DEFAULT TRUE,
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

-- 2. TABLA DE PRODUCTOS
-- El inventario. El precio es SIEMPRE en USD. 
-- El stock mínimo sirve para que la app nos avise: "Che, Carlitox, comprate más cemento".
CREATE TABLE productos (
  id           UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  nombre       TEXT NOT NULL,
  descripcion  TEXT,
  precio_usd   DECIMAL(12,2) NOT NULL, -- El valor real del fierro
  stock_actual INTEGER DEFAULT 0,
  stock_minimo INTEGER DEFAULT 0, -- Alerta de reposición
  categoria    TEXT,
  imagen_url   TEXT,
  activo       BOOLEAN DEFAULT TRUE, -- Por si dejamos de vender algo pero no queremos borrarlo
  created_at   TIMESTAMPTZ DEFAULT NOW()
);

-- 3. HISTORIAL DEL DÓLAR
-- Fundamental en Argentina. Guardamos la cotización de cada día. 
-- El "UNIQUE(fecha)" es para que no haya dos precios distintos el mismo día.
CREATE TABLE tipo_cambio (
  id           UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  fecha        DATE NOT NULL DEFAULT CURRENT_DATE,
  valor_blue   DECIMAL(12,2) NOT NULL, -- El que importa en el playón
  valor_oficial DECIMAL(12,2),          -- El que importa para ARCA
  fuente       TEXT DEFAULT 'dolarapi.com',
  created_at   TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(fecha)
);

-- 4. CAMIONES (LOGÍSTICA)
-- Para que el cuñado sepa dónde están los muchachos y en qué estado está el reparto.
CREATE TABLE camiones (
  id                  UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  nombre_chofer       TEXT NOT NULL,
  patente             TEXT,
  tipo                TEXT DEFAULT 'camion', -- Camión, F100, etc.
  estado              TEXT DEFAULT 'disponible', 
  ultima_lat          DECIMAL(10,8), -- Posición GPS para el Google Maps
  ultima_lng          DECIMAL(11,8),
  ultima_ubicacion_at TIMESTAMPTZ    -- Para saber si el chofer tiene el GPS apagado
);

-- 5. PRESUPUESTOS
-- La antesala de la venta. Se puede borrar, editar o vencer. 
-- Guardamos el dólar del momento para que el precio quede "congelado" si el cuñado quiere.
CREATE TABLE presupuestos (
  id               UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  cliente_id       UUID REFERENCES clientes(id),
  cotizacion_dolar DECIMAL(12,2),
  total_usd        DECIMAL(15,2),
  total_pesos      DECIMAL(15,2),
  descuento_pct    DECIMAL(5,2) DEFAULT 0,
  estado           TEXT DEFAULT 'borrador', -- Borrador, Aprobado, etc.
  notas            TEXT,
  created_at       TIMESTAMPTZ DEFAULT NOW(),
  vence_at         TIMESTAMPTZ 
);

-- 6. DETALLE DE PRESUPUESTOS (Items)
-- Como un presupuesto puede tener muchos productos, esta tabla guarda la lista.
CREATE TABLE presupuestos_items (
  id               UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  presupuesto_id   UUID REFERENCES presupuestos(id) ON DELETE CASCADE,
  producto_id      UUID REFERENCES productos(id),
  cantidad         INTEGER NOT NULL,
  precio_usd       DECIMAL(12,2) -- Precio al que se presupuestó ese día
);

-- 7. VENTAS (El registro final)
-- Cuando el cliente dice "sí, traeme las viguetas", el presupuesto se convierte en VENTA.
-- Acá separamos "OFICIAL" de "INFORMAL" para no mezclar las cajas.
CREATE TABLE ventas (
  id               UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  presupuesto_id   UUID REFERENCES presupuestos(id), 
  cliente_id       UUID REFERENCES clientes(id),
  camion_id        UUID REFERENCES camiones(id),
  tipo_venta       TEXT CHECK (tipo_venta IN ('oficial', 'informal')), -- ¡EL FILTRO CLAVE!
  cotizacion_dolar DECIMAL(12,2) NOT NULL,
  total_usd        DECIMAL(15,2),
  total_pesos      DECIMAL(15,2),
  descuento_pct    DECIMAL(5,2) DEFAULT 0,
  pagado           BOOLEAN DEFAULT FALSE,
  estado_entrega   TEXT DEFAULT 'pendiente', 
  notas            TEXT,
  created_at       TIMESTAMPTZ DEFAULT NOW()
);

-- 8. DETALLE DE VENTAS (Items)
-- Lo que efectivamente salió del depósito.
CREATE TABLE ventas_items (
  id                  UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  venta_id            UUID REFERENCES ventas(id) ON DELETE CASCADE,
  producto_id         UUID REFERENCES productos(id),
  cantidad            INTEGER NOT NULL,
  precio_unitario_usd DECIMAL(12,2)
);

-- =============================================
-- SEGURIDAD (RLS) - "Las llaves del boliche"
-- =============================================

-- Activamos la seguridad en todas las tablas
ALTER TABLE clientes ENABLE ROW LEVEL SECURITY;
ALTER TABLE productos ENABLE ROW LEVEL SECURITY;
ALTER TABLE camiones ENABLE ROW LEVEL SECURITY;
ALTER TABLE tipo_cambio ENABLE ROW LEVEL SECURITY;
ALTER TABLE presupuestos ENABLE ROW LEVEL SECURITY;
ALTER TABLE presupuestos_items ENABLE ROW LEVEL SECURITY;
ALTER TABLE ventas ENABLE ROW LEVEL SECURITY;
ALTER TABLE ventas_items ENABLE ROW LEVEL SECURITY;

-- REGLA 1: Vos (el Admin logueado) podés hacer de todo en todas las tablas.
CREATE POLICY "Admin full access" ON clientes FOR ALL TO authenticated USING (true);
CREATE POLICY "Admin full access" ON camiones FOR ALL TO authenticated USING (true);
CREATE POLICY "Admin full access" ON tipo_cambio FOR ALL TO authenticated USING (true);
CREATE POLICY "Admin full access" ON presupuestos FOR ALL TO authenticated USING (true);
CREATE POLICY "Admin full access" ON presupuestos_items FOR ALL TO authenticated USING (true);
CREATE POLICY "Admin full access" ON ventas FOR ALL TO authenticated USING (true);
CREATE POLICY "Admin full access" ON ventas_items FOR ALL TO authenticated USING (true);

-- REGLA 2: Los productos que están "activos" los puede ver cualquiera (para la futura web pública).
CREATE POLICY "Admin full access"  ON productos FOR ALL  TO authenticated USING (true);
CREATE POLICY "Public can view"    ON productos FOR SELECT TO anon USING (activo = true);

-- REGLA 3: El tipo de cambio es público, para que el cliente vea el dólar blue en la web.
CREATE POLICY "Public can read"    ON tipo_cambio FOR SELECT TO anon USING (true);
